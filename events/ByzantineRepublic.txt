# Election of 1939: Erdemir vs Lucenteforte vs Swiderska
country_event = {
	id = byz.8
	title = byz.8.t
	desc = byz.8.d
	picture = GFX_report_event_usa_election_generic

	trigger = {
		tag = BYZ
		communism > 0.5
		date > 1939.1.1
		date < 1940.1.1
		has_elections = yes
		NOT = { has_global_flag = byz_mullerist_coup  }
	}
	
	fire_only_once = yes
	
	option = { 	
		name = byz.8.a
		ai_chance = {
			factor = 75
		}		
		add_political_power = 150
		add_timed_idea = { idea = BYZ_electoral_supermajority days = 365 }
	}

	option = { 
		name = byz.8.b
		ai_chance = {
			factor = 15
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}		
		set_party_name = { ideology = communism long_name = BYZ_labour_party_long name = BYZ_labour_party }
		add_war_support = -0.05
		if = {
			limit = {
				has_idea = parthenon
			}
			add_political_power = 150
		}
		if = {
			limit = {
				NOT = { has_idea = parthenon }
			}
			add_ideas = parthenon
		}
		create_country_leader = {
			name = "Giancarlo Lucenteforte"
			desc = "POLITICS_GIANCARLO_LUCENTEFORTE_DESC"
			picture = "Portrait_USA_Generic_2.dds"
			expire = "1965.1.1"
			ideology = marxism
			traits = {
				staunch_constitutionalist
			}
		}
	}
	option = {
		name = byz.8.c
			ai_chance = {
				factor = 10
				modifier = {
					factor = 0
					is_historical_focus_on = yes
				}
			}
		if = {
			limit = {
				has_idea = democratic_drift
			}
			add_political_power = 150
		}
		if = {
			limit = {
				NOT = { has_idea = democratic_drift }
			}
			add_timed_idea = { idea = democratic_influence days = 180 }
		}
		create_country_leader = {
			name = "Anna Swiderska"
			desc = "POLITICS_ANNA_SWIDERSKA_DESC"
			picture = "Portrait_USA_Generic_2.dds"
			expire = "1965.1.1"
			ideology = marxism
			traits = {
			}
		}
	}
}
